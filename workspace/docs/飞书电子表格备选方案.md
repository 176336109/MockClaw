### 2.2 优化策略

#### 2.2.1 文本模拟表格
```python
def create_text_table(data, headers=None):
    """创建文本模拟表格"""
    if not data:
        return ""
    
    # 确定列宽
    col_widths = []
    if headers:
        for i, header in enumerate(headers):
            max_len = len(str(header))
            for row in data:
                if i < len(row):
                    max_len = max(max_len, len(str(row[i])))
            col_widths.append(max_len + 2)  # 加2个空格边距
    else:
        num_cols = max(len(row) for row in data) if data else 0
        col_widths = [15] * num_cols  # 默认宽度
    
    # 创建表格
    table_lines = []
    
    # 表头
    if headers:
        header_line = "|"
        for i, header in enumerate(headers):
            width = col_widths[i] if i < len(col_widths) else 15
            header_line += f" {str(header).ljust(width-2)} |"
        table_lines.append(header_line)
        
        # 分隔线
        separator = "|"
        for width in col_widths:
            separator += "-" * width + "|"
        table_lines.append(separator)
    
    # 数据行
    for row in data:
        row_line = "|"
        for i, cell in enumerate(row):
            width = col_widths[i] if i < len(col_widths) else 15
            row_line += f" {str(cell).ljust(width-2)} |"
        table_lines.append(row_line)
    
    return "\n".join(table_lines)
```

```python
class EnhancedMarkdownConverter:
    """增强的Markdown转换器"""
    
    def convert_table(self, markdown_table):
        """转换Markdown表格为优化格式"""
        lines = markdown_table.strip().split('\n')
        
        if len(lines) < 2:
            return markdown_table
        
        # 解析表头
        header = lines[0].strip('|').split('|')
        separator = lines[1]
        
        # 创建优化表格
        optimized = []
        
        # 添加样式
        optimized.append("**表格开始**")
        optimized.append("")  # 空行
        
        # 表头（加粗）
        header_line = "| " + " | ".join([f"**{h.strip()}**" for h in header]) + " |"
        optimized.append(header_line)
        optimized.append(separator)
        
        # 数据行
        for line in lines[2:]:
            if line.strip():
                optimized.append(line)
        
        optimized.append("")
        optimized.append("**表格结束**")
        
        return "\n".join(optimized)
    
    def create_data_summary(self, data):
        """创建数据摘要（替代复杂表格）"""
        summary = []
        
        if isinstance(data, list) and len(data) > 0:
            summary.append(f"**数据记录数**: {len(data)}")
            
            if len(data) <= 5:
                summary.append("**前几条记录**:")
                for i, item in enumerate(data[:5], 1):
                    summary.append(f"{i}. {str(item)[:100]}...")
            else:
                summary.append("**数据示例**:")
                for i in range(min(3, len(data))):
                    summary.append(f"{i+1}. {str(data[i])[:100]}...")
                summary.append(f"... 还有 {len(data)-3} 条记录")
        
        return "\n".join(summary)
```

```python
        response = request.execute()

            range=range_name,
            valueInputOption='USER_ENTERED',
```

```python
class AirtableIntegration:
    """Airtable集成"""
    
    def __init__(self, api_key, base_id):
        self.api_key = api_key
        self.base_id = base_id
        self.base_url = f"https://api.airtable.com/v0/{base_id}"
        self.headers = {
            'Authorization': f'Bearer {api_key}',
            'Content-Type': 'application/json'
        }
    
    def list_records(self, table_name, params=None):
        """列出记录"""
        import requests
        
        url = f"{self.base_url}/{table_name}"
        response = requests.get(url, headers=self.headers, params=params)
        response.raise_for_status()
        
        data = response.json()
        records = data.get('records', [])
        
        # 处理分页
        while 'offset' in data:
            params = params or {}
            params['offset'] = data['offset']
            response = requests.get(url, headers=self.headers, params=params)
            response.raise_for_status()
            data = response.json()
            records.extend(data.get('records', []))
        
        return records
    
    def create_record(self, table_name, fields):
        """创建记录"""
        import requests
        
        url = f"{self.base_url}/{table_name}"
        data = {
            'fields': fields
        }
        
        response = requests.post(url, headers=self.headers, json=data)
        response.raise_for_status()
        
        return response.json()
    
    def update_record(self, table_name, record_id, fields):
        """更新记录"""
        import requests
        
        url = f"{self.base_url}/{table_name}/{record_id}"
        data = {
            'fields': fields
        }
        
        response = requests.patch(url, headers=self.headers, json=data)
        response.raise_for_status()
        
        return response.json()
```

```python
import sqlite3
import json
from datetime import datetime

class LocalTableStore:
    """本地表格存储"""
    
    def __init__(self, db_path='tables.db'):
        self.db_path = db_path
        self.init_database()
    
    def init_database(self):
        """初始化数据库"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 创建表格元数据表
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS tables (
                id TEXT PRIMARY KEY,
                name TEXT NOT NULL,
                description TEXT,
                schema_json TEXT,
                created_at TIMESTAMP,
                updated_at TIMESTAMP
            )
        ''')
        
        # 创建数据表（动态创建）
        conn.commit()
        conn.close()
    
    def create_table(self, table_id, name, schema):
        """创建表格"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 保存表格元数据
        now = datetime.now()
        cursor.execute('''
            INSERT INTO tables (id, name, description, schema_json, created_at, updated_at)
            VALUES (?, ?, ?, ?, ?, ?)
        ''', (table_id, name, schema.get('description'), json.dumps(schema), now, now))
        
        # 动态创建数据表
        columns = []
        for field in schema.get('fields', []):
            field_name = field['name']
            field_type = self.map_field_type(field['type'])
            columns.append(f'{field_name} {field_type}')
        
        create_sql = f'''
            CREATE TABLE IF NOT EXISTS {table_id} (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                {', '.join(columns)},
                created_at TIMESTAMP,
                updated_at TIMESTAMP
            )
        '''
        
        cursor.execute(create_sql)
        conn.commit()
        conn.close()
        
        return table_id
    
    def map_field_type(self, field_type):
        """映射字段类型到SQLite类型"""
        mapping = {
            'text': 'TEXT',
            'number': 'REAL',
            'integer': 'INTEGER',
            'boolean': 'INTEGER',
            'date': 'TEXT',
            'datetime': 'TEXT',
            'json': 'TEXT'
        }
        return mapping.get(field_type, 'TEXT')
    
    def insert_record(self, table_id, data):
        """插入记录"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 构建插入语句
        columns = list(data.keys())
        placeholders = ['?' for _ in columns]
        values = list(data.values())
        
        # 添加时间戳
        columns.extend(['created_at', 'updated_at'])
        placeholders.extend(['?', '?'])
        now = datetime.now()
        values.extend([now, now])
        
        sql = f'''
            INSERT INTO {table_id} ({', '.join(columns)})
            VALUES ({', '.join(placeholders)})
        '''
        
        cursor.execute(sql, values)
        record_id = cursor.lastrowid
        
        conn.commit()
        conn.close()
        
        return record_id
    
    def query_records(self, table_id, conditions=None, limit=100, offset=0):
        """查询记录"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 构建查询语句
        sql = f'SELECT * FROM {table_id}'
        params = []
        
        if conditions:
            where_clauses = []
            for key, value in conditions.items():
                where_clauses.append(f'{key} = ?')
                params.append(value)
            
            if where_clauses:
                sql += ' WHERE ' + ' AND '.join(where_clauses)
        
        sql += f' LIMIT ? OFFSET ?'
        params.extend([limit, offset])
        
        cursor.execute(sql, params)
        rows = cursor.fetchall()
        
        # 获取列名
        column_names = [description[0] for description in cursor.description]
        
        conn.close()
        
        # 转换为字典列表
        records = []
        for row in rows:
            record = dict(zip(column_names, row))
            records.append(record)
        
        return records
```

### 4.2 JSON/YAML文件存储

#### 4.2.1 简单实现
```python
import json
import yaml
import os
from pathlib import Path

class FileTableStore:
    """文件表格存储"""
    
    def __init__(self, base_dir='data/tables'):
        self.base_dir = Path(base_dir)
        self.base_dir.mkdir(parents=True, exist_ok=True)
    
    def save_table(self, table_id, data, format='json'):
        """保存表格数据"""
        file_path = self.base_dir / f"{table_id}.{format}"
        
        if format == 'json':
            with open(file_path, 'w', encoding='utf-8') as f:
                json.dump(data, f, ensure_ascii=False, indent=2)
        elif format == 'yaml':
            with open(file_path, 'w', encoding='utf-8') as f:
                yaml.dump(data, f, allow_unicode=True)
        else:
            raise ValueError(f"不支持的格式: {format}")
        
        return str(file_path)
    
    def load_table(self, table_id, format='json'):
        """加载表格数据"""
        file_path = self.base_dir / f"{table_id}.{format}"
        
        if not file_path.exists():
            return None
        
        if format == 'json':
            with open(file_path, 'r', encoding='utf-8') as f:
                return json.load(f)
        elif format == 'yaml':
            with open(file_path, 'r', encoding='utf-8') as f:
                return yaml.safe_load(f)
        else:
            raise ValueError(f"不支持的格式: {format}")
    
    def export_to_csv(self, table_id, output_path=None):
        """导出为CSV"""
        data = self.load_table(table_id)
        if not data or 'records' not in data:
            return None
        
        import csv
        
        if output_path is None:
            output_path = self.base_dir / f"{table_id}.csv"
        
        records = data['records']
        if not records:
            return str(output_path)
        
        # 获取所有字段
        all_fields = set()
        for record in records:
            all_fields.update(record.keys())
        
        field_names = sorted(all_fields)
        
        with open(output_path, 'w', newline='', encoding='utf-8') as f:
            writer = csv.DictWriter(f, fieldnames=field_names)
            writer.writeheader()
            writer.writerows(records)
        
        return str(output_path)
```

```python
    def add_sheet(self, sheet_name, data, headers=None):
        """添加工作表"""
        if not self.workbook:
            raise ValueError("请先创建或加载工作簿")
        
        if sheet_name in self.workbook.sheetnames:
            ws = self.workbook[sheet_name]
        else:
            ws = self.workbook.create_sheet(title=sheet_name)
        
        # 写入表头
        if headers:
            for col_idx, header in enumerate(headers, 1):
                ws.cell(row=1, column=col_idx, value=header)
            start_row = 2
        else:
            start_row = 1
        
        # 写入数据
        for row_idx, row_data in enumerate(data, start_row):
            if isinstance(row_data, dict) and headers:
                # 字典数据，按表头顺序写入
                for col_idx, header in enumerate(headers, 1):
                    value = row_data.get(header, '')
                    ws.cell(row=row_idx, column=col_idx, value=value)
            elif isinstance(row_data, (list, tuple)):
                # 列表数据，直接写入
                for col_idx, value in enumerate(row_data, 1):
                    ws.cell(row=row_idx, column=col_idx, value=value)
            else:
                # 单个值
                ws.cell(row=row_idx, column=1, value=row_data)
        
        return ws
    
    def read_sheet(self, sheet_name, include_headers=True):
        """读取工作表"""
        if not self.workbook or sheet_name not in self.workbook.sheetnames:
            return None
        
        ws = self.workbook[sheet_name]
        data = []
        
        # 确定数据范围
        max_row = ws.max_row
        max_column = ws.max_column
        
        start_row = 1 if include_headers else 2
        
        for row in ws.iter_rows(min_row=start_row, max_row=max_row, 
                               min_col=1, max_col=max_column, values_only=True):
            # 过滤空行
            if any(cell is not None for cell in row):
                data.append(row)
        
        return data
```

#### 5.1.1 抽象层设计

```python
from abc import ABC, abstractmethod

class TableStorage(ABC):
    """表格存储抽象类"""
    
    @abstractmethod
    def create_table(self, name, schema):
        """创建表格"""
        pass
    
    @abstractmethod
    def insert_record(self, table_id, data):
        """插入记录"""
        pass
    
    @abstractmethod
    def query_records(self, table_id, conditions=None, limit=100, offset=0):
        """查询记录"""
        pass
    
    @abstractmethod
    def update_record(self, table_id, record_id, data):
        """更新记录"""
        pass
    
    @abstractmethod
    def delete_record(self, table_id, record_id):
        """删除记录"""
        pass

class TableStorageFactory:
    """表格存储工厂"""
    
    @staticmethod
    def create_storage(storage_type, **kwargs):
        """创建存储实例"""
        if storage_type == 'local_sqlite':
            return LocalTableStore(**kwargs)
        elif storage_type == 'google_sheets':
            return GoogleSheetsIntegration(**kwargs)
        elif storage_type == 'airtable':
            return AirtableIntegration(**kwargs)
        elif storage_type == 'file':
            return FileTableStore(**kwargs)
        elif storage_type == 'excel':
            return ExcelTableManager(**kwargs)
        else:
            raise ValueError(f"不支持的存储类型: {storage_type}")
```

#### 5.1.2 统一接口

```python
class UnifiedTableManager:
    """统一表格管理器"""
    
    def __init__(self, primary_storage=None, fallback_storage=None):
        self.primary_storage = primary_storage
        self.fallback_storage = fallback_storage
        self.storage_map = {}
    
    def register_table(self, table_id, storage_type, **storage_config):
        """注册表格到指定存储"""
        storage = TableStorageFactory.create_storage(storage_type, **storage_config)
        self.storage_map[table_id] = storage
        return storage
    
    def get_storage(self, table_id):
        """获取表格的存储实例"""
        if table_id in self.storage_map:
            return self.storage_map[table_id]
        elif self.primary_storage:
            return self.primary_storage
        else:
            raise ValueError(f"表格 '{table_id}' 未注册且无默认存储")
    
    def create_table(self, table_id, name, schema, storage_type='local_sqlite', **storage_config):
        """创建表格"""
        storage = self.register_table(table_id, storage_type, **storage_config)
        return storage.create_table(table_id, name, schema)
    
    def insert_record(self, table_id, data):
        """插入记录（带降级）"""
        try:
            storage = self.get_storage(table_id)
            return storage.insert_record(table_id, data)
        except Exception as e:
            if self.fallback_storage:
                # 降级到备用存储
                print(f"主存储失败，降级到备用存储: {e}")
                return self.fallback_storage.insert_record(table_id, data)
            else:
                raise
    
    def sync_table(self, source_table_id, target_table_id):
        """同步表格数据"""
        source_storage = self.get_storage(source_table_id)
        target_storage = self.get_storage(target_table_id)
        
        # 从源读取数据
        records = source_storage.query_records(source_table_id, limit=1000)
        
        # 写入目标
        for record in records:
            target_storage.insert_record(target_table_id, record)
        
        return len(records)
```

```python
import schedule
import time
from datetime import datetime

class DataSyncManager:
    """数据同步管理器"""
    
    def __init__(self, table_manager):
        self.table_manager = table_manager
        self.sync_jobs = {}
    
    def setup_sync(self, source_table_id, target_table_id, interval_minutes=60):
        """设置同步任务"""
        job_id = f"{source_table_id}_to_{target_table_id}"
        
        def sync_job():
            print(f"[{datetime.now()}] 开始同步 {source_table_id} -> {target_table_id}")
            try:
                count = self.table_manager.sync_table(source_table_id, target_table_id)
                print(f"[{datetime.now()}] 同步完成，共 {count} 条记录")
            except Exception as e:
                print(f"[{datetime.now()}] 同步失败: {e}")
        
        # 安排定时任务
        schedule.every(interval_minutes).minutes.do(sync_job)
        self.sync_jobs[job_id] = sync_job
        
        # 立即执行一次
        sync_job()
        
        return job_id
    
    def start_sync_loop(self):
        """启动同步循环"""
        print("启动数据同步循环...")
        while True:
            schedule.run_pending()
            time.sleep(1)
```

```python
class ChangeListener:
    """变更监听器"""
    
    def __init__(self):
        self.listeners = {}
        self.change_log = []
    
    def register_listener(self, table_id, callback):
        """注册变更监听器"""
        if table_id not in self.listeners:
            self.listeners[table_id] = []
        self.listeners[table_id].append(callback)
    
    def notify_change(self, table_id, operation, record_id, data=None):
        """通知变更"""
        change_event = {
            'timestamp': datetime.now(),
            'table_id': table_id,
            'operation': operation,
            'record_id': record_id,
            'data': data
        }
        
        self.change_log.append(change_event)
        
        # 通知监听器
        if table_id in self.listeners:
            for callback in self.listeners[table_id]:
                try:
                    callback(change_event)
                except Exception as e:
                    print(f"监听器执行失败: {e}")
        
        return change_event
    
    def get_recent_changes(self, table_id=None, limit=100):
        """获取最近变更"""
        changes = self.change_log
        
        if table_id:
            changes = [c for c in changes if c['table_id'] == table_id]
        
        return changes[-limit:]
```
